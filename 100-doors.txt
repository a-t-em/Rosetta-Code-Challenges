function getFinalOpenedDoors(n_doors) {
  var doors = [];
  doors.length = 100;
  doors.fill(0);
  var open_doors = [];
  for (let i=2; i<=100; i++) {
    for (let j=i-1; j<=100; j+=i) {
      if (doors[j] == 1) {
        doors[j] = 0;
      } else {
        doors[j] = 1;
      }
      }
  }
  for (let i=0; i<100; i++) {
    if (doors[i] == 0) {
      open_doors.push(i+1);
    }
  }
   return open_doors;
}
